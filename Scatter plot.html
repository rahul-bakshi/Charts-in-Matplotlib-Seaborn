plt.figure(figsize = (12,8))
scatter = plt.scatter(df['Area (ft.)'],
            df['Price'], alpha = 0.6, c=df['Building Type'], cmap = 'viridis')
plt.legend(*scatter.legend_elements(), #pointer to the legend elements of the scatter
           loc = 'upper left', 
           title = "Builidng Type") 
plt.title("Relationship between Area and Price",fontsize=14,fontweight="bold")
plt.xlabel("Area in sq.ft")
plt.ylabel("Price($)")
plt.show()
